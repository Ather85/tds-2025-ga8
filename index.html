Code: Compute YoY Growth


import pandas as pd

df = pd.read_csv("quarterly.csv", parse_dates=["date"])
latest = df[df.date == "2025-09-30"]
prior = df[df.date == "2024-09-30"]

yoy_growth = (latest.revenue.values[0] - prior.revenue.values[0]) / prior.revenue.values[0] * 100
print(f"YoY revenue growth: {yoy_growth:.1f}%")
